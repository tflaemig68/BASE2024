90/**
91 * @ingroup exti1
92 * Sets the selected EXTI according to the port/pin
93 *
94 * @param *port : Pointer to the GPIO port
95 * @param pin : Number of the pin which shall be used as EXTI
96 * @return EXTI_IRQ_NUM
97 *
98 * <br>
99 * <b>Affected register and bit(s)</b><br>
100 * <table>
101 * <tr>
102 * <th>Register</th>
103 * <th>Bit name</th>
104 * <th>Bit(s)</th>
105 * </tr>
106 * <tr>
107 * <td>EXTICR4/3/2/1</td>
108 * <td rowspan="1">EXTI0...EXTI15</td>
109 * <td rowspan="1">15...0</td>
110 * </tr>
111 * </table>
112 */
113EXTI_IRQ_NUM extiConfigIrq(GPIO_TypeDef *port, PIN_NUM_t pin)
114{
115 uint8_t index = 0;
116 uint8_t shift = 0;
117 uint8_t mask = 0;
118
119 if (gpioVerifyPort(port) != true)
120 {
121 return GPIO_INVALID_PORT;
122 }
123 if ((gpioVerifyPin(pin)) != true)
124 {
125 return GPIO_INVALID_PIN;
126 }
127
128 switch ((uint8_t) pin)
129 {
130 case PIN0:
131 case PIN1:
132 case PIN2:
133 case PIN3:
134 shift = pin * 4; // Every pin uses four bits
135 index = 0; // Set the array-index of SYSCFG->EXTICR[]
136 break;
137
138 case PIN4:
139 case PIN5:
140 case PIN6:
141 case PIN7:
142 shift = (pin - 4) * 4;
143 index = 1;
144 break;
145
146 case PIN8:
147 case PIN9:
148 case PIN10:
149 case PIN11:
150 shift = (pin - 8) * 4;
151 index = 2;
152 break;
153
154 case PIN12:
155 case PIN13:
156 case PIN14:
157 case PIN15:
158 shift = (pin - 12) * 4;
159 index = 3;
160 break;
161 }
162
163 if (GPIOA == port)
164 {
165 mask = PORT_A; // Inversion of the port mask
166 }
167 else if (GPIOB == port)
168 {
169 mask = PORT_B;
170 }
171 else if (GPIOC == port)
172 {
173 mask = PORT_C;
174 }
175 else if (GPIOD == port)
176 {
177 mask = PORT_D;
178 }
179 else if (GPIOE == port)
180 {
181 mask = PORT_E;
182 }
183 else if (GPIOF == port)
184 {
185 mask = PORT_F;
186 }
187 else if (GPIOG == port)
188 {
189 mask = PORT_G;
190 }
191 else if (GPIOH == port)
192 {
193 mask = PORT_H;
194 }
195
196 SYSCFG->EXTICR[index] |= (mask << shift);
197
198 return EXTI_OK;
199}

265/**
266 * @ingroup exti1
267 * Sets the trigger mode of the selected EXTI.
268 *
269 * @param irqNum : One from 23 EXTI IRQs
270 * @param trigger : rising only/falling only/rising + falling
271 * @return EXTI_IRQ_NUM
272 *
273 * <br>
274 * <b>Affected register and bit(s)</b><br>
275 * <table>
276 * <tr>
277 * <th>Register</th>
278 * <th>Bit name</th>
279 * <th>Bit(s)</th>
280 * </tr>
281 * <tr>
282 * <td>RTSR</td>
283 * <td rowspan="1">TR22...TR0</td>
284 * <td rowspan="1">22...0</td>
285 * </tr>
286 * <tr>
287 * <td>FTSR</td>
288 * <td rowspan="1">TF22...TF0</td>
289 * <td rowspan="1">22...0</td>
290 * </tr>
291 * </table>
292 */
293EXTI_IRQ_NUM extiSetTriggerEdge(EXTI_IRQ_NUM irqNum, EXTI_TRIGGER trigger)
294{
295 if (extiVerifyIrqNum(irqNum) != true)
296 {
297 return EXTI_INVALID_IRQNUM;
298 }
299
300 if (extiVerifyTrigger(trigger) != true)
301 {
302 return EXTI_INVALID_TRIGGER;
303 }
304
305 if (RISING_EDGE == trigger)
306 {
307 EXTI->RTSR |= 1 << irqNum; // Enable rising edge
308 EXTI->FTSR &= ~(1 << irqNum); // Disable falling edge
309 }
310 else if (FALLING_EDGE == trigger)
311 {
312 EXTI->FTSR |= 1 << irqNum; // Enable falling edge
313 EXTI->RTSR &= ~(1 << irqNum); // Disable rising edge
314 }
315 else if (RISING_AND_FALLING == trigger)
316 {
317 EXTI->RTSR |= 1 << irqNum; // Enable rising edge
318 EXTI->FTSR |= 1 << irqNum; // Enable falling edge
319 }
320 return EXTI_OK;
321}
